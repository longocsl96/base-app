# base-app

> Base app

## Usage

```jsx
import React, { Component } from 'react'

import BaseApp from 'base-app'
import 'base-app/dist/index.css'

class Example extends Component {
  render() {
    return <BaseApp
        history={history}
        appReducer={emails}
        message={message}
        navigationConfig={navigationConfig}
      />
  }
}
```
